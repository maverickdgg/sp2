700  glUniform3fv(m_parameters[U_MATERIAL_DIFFUSE], 1, &mesh->material.kDiffuse.r);
701  glUniform3fv(m_parameters[U_MATERIAL_SPECULAR], 1, &mesh->material.kSpecular.r);
702  glUniform1f(m_parameters[U_MATERIAL_SHININESS], mesh->material.kShininess);
703  }
704  else
705  {
706  glUniform1i(m_parameters[U_LIGHTENABLED], 0);
707  }

709  {
710  glUniform1i(m_parameters[U_COLOR_TEXTURE_ENABLED], 1);
711  glActiveTexture(GL_TEXTURE0);
712  glBindTexture(GL_TEXTURE_2D, mesh->textureID);
713  glUniform1i(m_parameters[U_COLOR_TEXTURE], 0);

717  glUniform1i(m_parameters[U_COLOR_TEXTURE_ENABLED], 0);

720  if (mesh->textureID > 0)
721  {
722  glBindTexture(GL_TEXTURE_2D, 0);
723  }
